#include<stdio.h>

main()
{
  int n,num,i,pos_sum = 0,neg_sum = 0;
  
  printf("Enter the value of num:");
  scanf("%d",&num);
  
  for(i = 1;i<=num;i++) {
  	
  	printf("Enter %d number",i);
  	scanf("%d",&n);
   if(n>0)
   pos_sum = pos_sum+n;
   else 
   neg_sum = neg_sum+n;
}

  printf("Sum of all positive numbers = %d\n",pos_sum);
  printf("Sum of all negative numbers = %d\n",neg_sum);	
	return 0;
}
