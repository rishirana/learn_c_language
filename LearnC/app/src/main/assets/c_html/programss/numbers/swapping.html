#include<stdio.h>

main()
{
 
 int a,b,temp;
  
  printf("Enter the value of a and b\n");
  scanf("%d %d",&a,&b);
  
  printf("Before swapping\n A = %d\nB = %d\n",a,b);
  
  temp = a;
  a = b;
  b = temp;
  
  printf("After swapping\n A = %d\nB = %d\n",a,b);
 
	return 0;
}
