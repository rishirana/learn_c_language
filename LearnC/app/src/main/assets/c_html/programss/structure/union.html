#include <stdio.h>

void main(){
	union number
	{
		int n1;
		float n2;
		
	};
	union number x;
	
	printf("Enter the value of n1 :");
	scanf("%d",&x.n1);
	printf("Enter the value of n2 :");
	scanf("%f",&x.n2);
	printf("Value of n2 : %.2f\n",x.n2);
}
Output:
Enter the value of n1 :4
Enter the value of n2 :7
Value of n2 : 7.00
