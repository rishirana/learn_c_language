#include <stdio.h>

int main()
{
    int i, j, N;

    //Reads total number of rows from user
    printf("Enter rows: ");
    scanf("%d", &N);

    for(i=1; i<=N; i++)
    {
        //Print trailing spaces
        for(j=1; j<=N - i; j++)
        {
            printf(" ");
        }

        //Print stars after spaces
        for(j=1; j<=N; j++)
        {
            printf("*");
        }

        //Moves to the next line/row
        printf("\n");
    }

    return 0;
}
