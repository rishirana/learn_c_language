#include <stdio.h>


struct student
{
   char name[32];
   int roll;
    //array of marks
    int marks[3];
    int total;
    float avg;
};

void main()
{
	int i;
	struct student s;
	printf("\n\nEnter rollno:");
	scanf("%d",&s.roll);
	
	printf("\n\nEnter Name:");
	scanf("%s",&s.name);
	
	s.total = 0;
	 
	for(i=0;i<3;i++)
	{
	printf("\n\nEnter marks %d",i+1);
	scanf("%d",&s.marks[i]);
	s.total+=s.marks[i];
		
	 } 
	 
	 s.avg= s.total/3;
	 
	 printf("\nRoll:%d",s.roll);
	 printf("\nName:%s",s.name);
	 printf("\nTotal:%d",s.total);
	 printf("\nAverage:%d",s.avg);
	
}
